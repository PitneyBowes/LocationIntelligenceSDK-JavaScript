 23 /**
 24  * Set API URL to get Route by address
 25  * @param Object {startAddress: string(required), endAddress: string(required), db: string(optional), country: string(optional), intermediateAddresses: string(optional), oip: boolean(optional),  destinationSrs: string(optional), optimizeBy: string(optional), returnDistance: boolean(optional), distanceUnit: string(optional), returnTime: boolean(optional), timeUnit: string(optional), language: string(optional), directionsStyle: string(optional), segmentGeometryStyle: string(optional), primaryNameOnly: boolean(optional), majorRoads: boolean(optional), returnDirectionGeometry: boolean(optional), historicTrafficTimeBucket: string(optional)}
 26  * @callback {callback: User defined callback} (optional) or
 27  * @return response object
 28  */
 29 GEOAPIS_V1.geoRoute.prototype.getRouteByAddress = function(params, callback){
 30 	var apiUrl =  '/georoute/v1/route/byaddress?startAddress='+encodeURIComponent(params.startAddress)+'&endAddress='+encodeURIComponent(params.endAddress);
 31 	optionalList = ['db', 'country', 'intermediateAddresses', 'oip', 'destinationSrs', 'optimizeBy', 'returnDistance', 'distanceUnit', 'returnTime', 'timeUnit', 'language', 'directionsStyle', 'segmentGeometryStyle', 'primaryNameOnly', 'majorRoads', 'returnDirectionGeometry', 'historicTrafficTimeBucket'];
 32 	apiUrl = this.apiGetUrl(params, optionalList, apiUrl);
 33 	if(callback !== undefined){
 34 		this.callApiAsync(apiUrl, callback);
 35 	}
 36 	else{
 37 		this.callApi(apiUrl);
 38 		return this.response;
 39 	}
 40 };
