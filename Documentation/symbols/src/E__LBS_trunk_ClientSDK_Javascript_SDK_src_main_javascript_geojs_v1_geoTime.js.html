 12 /**
 13  * A geoTime object
 14  * @namespace
 15  */
 16 GEOAPIS_V1.geoTime = function(accessToken){
 17 	GEOAPIS_V1.baseService.call(this, accessToken);
 18 };
 19 /**
 20  * geoTime inherits baseService
 21  * @namespace
 22  */
 23 GEOAPIS_V1_INHERIT(GEOAPIS_V1.baseService, GEOAPIS_V1.geoTime);
 24 /**
 25  * Set API URL to get timezone
 26  * @param Object {address: free-form text(required), timestamp: float(required)}
 27  * @callback {callback: User defined callback} (optional) or
 28  * @return response object
 29  */
 30 GEOAPIS_V1.geoTime.prototype.getTimezoneByAddress = function(params, callback){
 31 	var apiUrl = '/geotime/v1/timezone/byaddress?address='+encodeURIComponent(params.address)+'×tamp='+encodeURIComponent(params.timestamp);
 32 	if(callback !== undefined){
 33 		this.callApiAsync(apiUrl, callback);
 34 	}
 35 	else{
 36 		this.callApi(apiUrl);
 37 		return this.response;
 38 	}
 39 };
 40 /**
 41  * Set API URL to get timezone
 42  * @param Object {latitude: float(required), longitude: float(required), timestamp: float(required)}
 43  * @callback {callback: User defined callback} (optional) or
 44  * @return response object
 45  */
 46 GEOAPIS_V1.geoTime.prototype.getTimezoneByLocation = function(params, callback){
 47 	var apiUrl = '/geotime/v1/timezone/bylocation?latitude='+encodeURIComponent(params.latitude)+'&longitude='+encodeURIComponent(params.longitude)+'×tamp='+encodeURIComponent(params.timestamp);
 48 	if(callback !== undefined){
 49 		this.callApiAsync(apiUrl, callback);
 50 	}
 51 	else{
 52 		this.callApi(apiUrl);
 53 		return this.response;
 54 	}
 55 };
