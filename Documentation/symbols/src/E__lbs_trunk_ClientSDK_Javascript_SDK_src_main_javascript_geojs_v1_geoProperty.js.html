 11 /**
 12  * A geoProperty object
 13  * @namespace
 14  */
 15  GEOAPIS_V1.geoProperty = function(accessToken){
 16 	GEOAPIS_V1.baseService.call(this, accessToken);
 17 };
 18 /**
 19  * geoProperty inherits baseService
 20  * @namespace
 21  */
 22 GEOAPIS_V1_INHERIT(GEOAPIS_V1.baseService, GEOAPIS_V1.geoProperty);
 23 /**
 24  * Set API URL to get Geo Property by Address
 25  * @param Object {category: string(required), address: string(required)}
 26  * @callback {callback: User defined callback} (optional) or
 27  * @return response object
 28  */
 29 GEOAPIS_V1.geoProperty.prototype.getGeoPropertyByAddress = function(params, callback){
 30 	var apiUrl =  '/geoproperty/v1/'+encodeURIComponent(params.category)+'/attributes/byaddress?address='+encodeURIComponent(params.address);
 31 	if(callback !== undefined){
 32 		this.callApiAsync(apiUrl, callback);
 33 	}
 34 	else{
 35 		this.callApi(apiUrl);
 36 		return this.response;
 37 	}
 38 };
 39 /**
 40  * Set API URL to post geoProperty by Address Batch
 41  * @param Object {category: string(required),addresses: The address or addresses to get GeoProperty (required)}
 42  * @callback {callback: User defined callback} (optional) or
 43  * @return response object
 44  */
 45 GEOAPIS_V1.geoProperty.prototype.getGeoPropertyByAddressBatch = function(params, callback){
 46 	var apiUrl =  '/geoproperty/v1/'+encodeURIComponent(params.category)+'/attributes/byaddress';
 47 	postData = '{"addresses":'+params.addresses+'}';
 48 	if(callback !== undefined){
 49 		this.callPostApiAsync(apiUrl, postData, callback);
 50 	}
 51 	else{
 52 		this.callPostApi(apiUrl, postData);
 53 		return this.response;
 54 	}
 55 };
