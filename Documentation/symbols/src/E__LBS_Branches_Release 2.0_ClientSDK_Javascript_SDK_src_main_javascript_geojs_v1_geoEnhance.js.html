 24 /**
 25  * Set API URL to get address
 26  * @param Object {latitude: float(required), longitude: float(required), searchRadius: number(optional), searchRadiusUnit: string(optional)}
 27  * @param String callback method name (optional)
 28  * @return response object or calls callback
 29  */
 30 GEOAPIS_V1.geoEnhance.prototype.getAddress = function(params, callback){
 31 	var apiUrl = '/geoenhance/v1/address/bylocation?latitude='+encodeURIComponent(params.latitude)+'&longitude='+encodeURIComponent(params.longitude);
 32 	var optionalList = ['searchRadius', 'searchRadiusUnit'];
 33 	apiUrl = this.apiGetUrl(params, optionalList, apiUrl);
 34 	if(callback !== undefined){
 35 		this.callApiAsync(apiUrl, callback);
 36 	}
 37 	else{
 38 		this.callApi(apiUrl);
 39 		return this.response;
 40 	}
 41 };
