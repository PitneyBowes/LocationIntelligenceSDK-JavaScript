 30 /**
 31  * Call API URL and set response property
 32  * @param string relative apiUrl
 33  */
 34 GEOAPIS_V1.baseService.prototype.callApi = function(apiUrl){
 35 	var request=null;
 36 	this.response = {};
 37 	this.response.httpResponse = {};
 38 	try {
 39 		request = $.ajax({
 40 			url: this.apiAddress+apiUrl,			
 41 			type: 'GET',
 42 			async: false,
 43 			headers: {
 44 				'Content-type': 'application/json',
 45 				'Accept': 'application/json',
 46 				'Authorization': this.accessToken
 47 			}
 48 		});		
 49 		this.response.response = JSON.parse(request.responseText);
 50 		if(request.status === 200){
 51 			this.response.status = 'success';
 52 		}
 53 		else{
 54 			this.response.status = 'failed';
 55 		}
 56 	} catch(e){
 57 		this.response.status = 'failed';
 58 		if(request.responseText !== ""){
 59 			this.response.response = request.responseText;
 60 		}
 61 		else{
 62 			this.response.response = {};
 63 			this.response.response.errors = [];
 64 			this.response.response.errors.push({errorCode:"GE0001",errorDescription:"Internal server error encountered."});
 65 		}
 66 	}
 67 	this.response.httpResponse.status = request.status;
 68 	this.response.httpResponse.statusText = request.statusText;
 69 };

230 GEOAPIS_V1.baseService.prototype.apiGetUrl = function(params, optionalList, apiUrl){
231 	for(var key in params){
232 		if(optionalList.indexOf(key) >= 0){
233 			if(params[key] !== null && params[key] !== undefined){
234 				apiUrl += this.queryDelimiter(apiUrl)+key+'='+encodeURIComponent(params[key]);
235 			}
236 		}
237 	}
238 	return apiUrl;
239 };
240 GEOAPIS_V1.baseService.prototype.queryDelimiter = function(requestUri){
241 	var queryDelimiter = '&';
242 	if(requestUri.search('\\?') < 0){
243 		queryDelimiter = '?';
244 	}
245 	return queryDelimiter;
