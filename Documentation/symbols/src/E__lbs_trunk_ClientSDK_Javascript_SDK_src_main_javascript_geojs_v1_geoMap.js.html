 11 /**
 12  * A geoMap object
 13  * @namespace
 14  */
 15  GEOAPIS_V1.geoMap = function(apiKey){
 16 	GEOAPIS_V1.baseService.call(this, '');
 17 	this.apiKey = apiKey;
 18 };
 19 /**
 20  * geoMap inherits baseService
 21  * @namespace
 22  */
 23 GEOAPIS_V1_INHERIT(GEOAPIS_V1.baseService, GEOAPIS_V1.geoMap);
 24 /**
 25  * Get GeoMap
 26  * @param Object initialView: this object contains information about initial view. It has three properties {latitude: float, longitude: float, zoom: number} (required)
 27  * @param String eleId: HTML element Id to render map (required)
 28  * @return map object
 29  */
 30 GEOAPIS_V1.geoMap.prototype.getGeoMap = function(initialView, eleId){
 31 	$('#'+eleId).html('');
 32 	$( '<div></div>', {
 33 		'id': 'geoAPI-geoMap-raster'
 34 	}).appendTo('#'+eleId);
 35 	$( '<div></div>', {
 36 		'id': 'geoAPI-geoMap-raster-logo'
 37 	}).appendTo('#'+eleId);
 38 	$('#geoAPI-geoMap-raster').css('position', 'absolute');
 39 	$('#geoAPI-geoMap-raster').css('width', '100%');
 40 	$('#geoAPI-geoMap-raster').css('height', '100%');
 41 	$( '<img/>', {
 42 		'src': 'https://developer2.pitneybowes.com/en/pitneyboweslogo.png'
 43 	}).appendTo('#geoAPI-geoMap-raster-logo');
 44 	$('#geoAPI-geoMap-raster-logo').css('position', 'absolute');
 45 	$('#geoAPI-geoMap-raster-logo').css('bottom', '0px');
 46 	$('#geoAPI-geoMap-raster-logo').css('z-index', '4');
 47 	var map = L.map('geoAPI-geoMap-raster');
 48 	map.setView([initialView.latitude, initialView.longitude], initialView.zoom);
 49 	return map;
 50 };
 51 /**
 52  * Get GeoMap (Raster)
 53  * @param Object mapObj: map object to render raster tile layer (required)
 54  * @param String mapStyle: map style bronze/iron/steel (optional)
 55  * @return map object
 56  */
 57 GEOAPIS_V1.geoMap.prototype.getGeoMapRaster = function(mapObj, mapStyle){
 58 	var tmpMapStyle = 'bronze',
 59 	tileLayerUrl,
 60 	tileLayer;
 61 	if(mapStyle == 'bronze' || mapStyle == 'iron' || mapStyle == 'steel'){
 62 		tmpMapStyle = mapStyle;
 63 	}
 64 	tileLayerUrl = this.apiAddress+ '/geomap/v1/tile/osm/{z}/{x}/{y}.png?api_key=' +this.apiKey+ '&theme=' +tmpMapStyle;
 65 	tileLayer = L.tileLayer(tileLayerUrl, {attribution: '<a target="_blank" href="https://carto.com/attributions">© CARTO</a> | <a target="_blank" href="http://www.openstreetmap.org/copyright">© OpenStreetMap contributors</a>'}).addTo(mapObj);
 66 	return mapObj;
 67 };
 68 /**
 69  * Get GeoMap (Vector)
 70  * @param Object mapObj: map object to render vector tile layer (required)
 71  * @param String mapStyle: map style bronze/iron/steel (optional)
 72  * @return map object
 73  */
 74 GEOAPIS_V1.geoMap.prototype.getGeoMapVector = function(mapObj, mapStyle){
 75 	var tmpMapStyle = 'bronze',
 76 	tileLayer;
 77 	if(mapStyle == 'bronze' || mapStyle == 'iron' || mapStyle == 'steel'){
 78 		tmpMapStyle = mapStyle;
 79 	}
 80 	tileLayer = Tangram.leafletLayer({
 81 		scene: 'resources/'+tmpMapStyle+'.yaml',
 82 		attribution: '<a target="_blank" href="https://carto.com/attributions">© CARTO</a> | <a target="_blank" href="http://www.openstreetmap.org/copyright">© OpenStreetMap contributors</a> | <a target="_blank" href="http://mapzen.com">© Mapzen</a>',
 83 		introspection: true
 84 	}).addTo(mapObj);
 85 	return mapObj;
 86 };
