 23 /**
 24  * Set API URL to get tax rate by address
 25  * @param Object {taxType: Auto/General/Medical/Construction (required), address: free-form text} (required)
 26  * @param String callback method name (optional)
 27  * @return response object or calls callback
 28  */
 29 GEOAPIS_V1.geoTax.prototype.getTaxRateByAddress = function(params, callback){
 30 	var apiUrl = '/geotax/v1/taxrate/'+encodeURIComponent(params.taxType)+'/byaddress?address='+encodeURIComponent(params.address);
 31 	if(callback !== undefined){
 32 		this.callApiAsync(apiUrl, callback);
 33 	}
 34 	else{
 35 		this.callApi(apiUrl);
 36 		return this.response;
 37 	}
 38 };
